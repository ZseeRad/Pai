Napisz procedure przyjmującą jako parametry imie,nazwisko kierowcy
i wyświetlająca następujące informację:
Imie nazwisko kierowcy,
Minimalną,max ilość pktów, średnią ilość punktów
w ilu wyscigach bral udzial

CREATE PROCEDURE wys15(u_imie VARCHAR(30), u_nazwisko VARCHAR(30))
BEGIN

DECLARE pkt_max VARCHAR(20);
DECLARE pkt_min VARCHAR(20);
DECLARE pkt_srd VARCHAR(20);
DECLARE ile_wys INT;
DECLARE kierowca_id VARCHAR(20);

SELECT id_kierowcy INTO kierowca_id FROM kierowcy WHERE imie=u_imie AND nazwisko=u_nazwisko;

SELECT MAX(punkty),MIN(punkty),AVG(punkty),COUNT(*) INTO pkt_max,pkt_min,pkt_srd,ile_wys FROM wyniki WHERE id_kierowcy=kierowca_id;
SELECT imie,nazwisko,pkt_max,pkt_min,pkt_srd,ile_wys FROM kierowcy WHERE imie=u_imie AND nazwisko=u_nazwisko;

END;
